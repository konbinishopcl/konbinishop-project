      user      User     @relation("UserMemberships", fields: [userId], references: [id], onDelete: Cascade)
      org       User     @relation("OrgMembers",      fields: [orgId],  references: [id], onDelete: Cascade)
      createdAt DateTime @default(now())

      @@unique([userId, orgId])
      @@index([orgId])
      @@index([userId])

      @@index([orgId])
      @@index([token])
    }
