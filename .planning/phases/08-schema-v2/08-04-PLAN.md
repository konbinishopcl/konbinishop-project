    **(c) Modelo Notification:**

    ```prisma
    // Mensaje interno generado automáticamente por el sistema.
    // El destinatario puede ser un user PERSON (userId) o una org (orgId).
    // Exactamente uno de userId / orgId debe estar presente — validado en service layer.
    model Notification {
      id        Int              @id @default(autoincrement())
      type      NotificationType
      title     String
      body      String?          @db.Text
      payload   Json             @default("{}")    // datos extra (ej: eventId, fromUserId)
      read      Boolean          @default(false)

      user      User?            @relation("UserNotifications", fields: [userId], references: [id], onDelete: Cascade)
      userId    Int?
      org       User?            @relation("OrgNotifications",  fields: [orgId],  references: [id], onDelete: Cascade)
      orgId     Int?

      createdAt DateTime         @default(now())

      @@index([userId, read])
      @@index([orgId, read])
      @@index([createdAt])
    }
    ```

    **(d) Modelo SavedEvent:**

    ```prisma
    // Favoritos de eventos — un user puede guardar muchos eventos, sin duplicados.
    model SavedEvent {
      userId    Int
      eventId   Int
      user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
      event     Event    @relation(fields: [eventId], references: [id], onDelete: Cascade)
      createdAt DateTime @default(now())

      @@unique([userId, eventId])
      @@index([userId])
      @@index([eventId])
    }
    ```

    **(e) Modelo Subscription:**

    ```prisma
    // Suscripción mensual. Puede pertenecer a un user PERSON o a una org (XOR).
    // creditsUsed se resetea al inicio de cada ciclo; no se acumulan.
    model Subscription {
      id              Int                @id @default(autoincrement())
      status          SubscriptionStatus @default(ACTIVE)

      user            User?              @relation("UserSubscription", fields: [userId], references: [id], onDelete: Cascade)
      userId          Int?               @unique  // máximo una sub activa por user
      org             User?              @relation("OrgSubscription",  fields: [orgId],  references: [id], onDelete: Cascade)
      orgId           Int?               @unique  // máximo una sub activa por org

      cycleStart      DateTime
      cycleEnd        DateTime
      creditsUsed     Int                @default(0)
      creditsTotal    Int                @default(10)  // copiado desde Settings.SUBSCRIPTION_CREDITS al activar

      cancelledAt     DateTime?
      createdAt       DateTime           @default(now())
      updatedAt       DateTime           @updatedAt

      @@index([status])
      @@index([cycleEnd])
    }
    ```

    **(f) Modelo Transfer:**

    ```prisma
    // Transferencia polymorphic de un ítem (Event/Spot/Hero/Article) entre cuentas.
    // No hay FKs múltiples al ítem — itemType + itemId; la validación de existencia
    // se hace en service layer (patrón replicado de AuditLog).
    model Transfer {
      id          Int              @id @default(autoincrement())
      itemType    TransferItemType
      itemId      Int

      fromUser    User             @relation("TransferFrom", fields: [fromUserId], references: [id], onDelete: Cascade)
      fromUserId  Int
      toOrg       User             @relation("TransferTo",   fields: [toOrgId],    references: [id], onDelete: Cascade)
      toOrgId     Int

      status      TransferStatus   @default(PENDING)
      reason      String?          // motivo si REJECTED
      resolvedBy  Int?             // userId del OWNER que aceptó/rechazó, o admin
      resolvedAt  DateTime?

      createdAt   DateTime         @default(now())
      updatedAt   DateTime         @updatedAt

      @@index([toOrgId, status])
      @@index([fromUserId])
      @@index([itemType, itemId])
    }
    ```
