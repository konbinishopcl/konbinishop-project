    // Decisión locked KEY #2: CrmEntry es un modelo independiente. ContactMessage

    // CrmEntry con sourceType=CONTACT, sourceId=contactMessage.id.

      id          Int       @id @default(autoincrement())
      type        CrmType
      stage       CrmStage  @default(NEW)
      stageReason String?               // requerido si stage=LOST (validación service layer)

      // Polymorphic source — apunta a ContactMessage o ServiceRequest según `type`.

    }
