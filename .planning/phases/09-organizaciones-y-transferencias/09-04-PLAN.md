<objective>
Crear el módulo `transfers` que permite mover un Event/Spot/Hero/Article de una cuenta personal a una organización. La transferencia es **polimórfica** (sin FKs múltiples — patrón replicado de AuditLog): el modelo `Transfer` ya existe en schema con `itemType: TransferItemType + itemId: Int`.

Auto-aprobación: si el caller es OWNER de la org destino → `status: AUTO_ACCEPTED` y ownership update inmediato. Si es MEMBER → `status: PENDING` y email al OWNER. Admin tiene endpoint paralelo `POST /admin/transfers` con `status: ADMIN_FORCED`.

Purpose: ORG-04. Cierra el flujo de gobernanza de contenido — permite a un usuario que publicó algo en su cuenta personal moverlo a una organización (caso típico: "lo creé yo, pero pertenece al colectivo").

Output: Módulo `transfers` con 5 endpoints REST, validación polimórfica de ownership por tipo de ítem, integración con OrgContextGuard (09-02) y MailService extendido.
</objective>

    import { Body, Controller, Get, Param, ParseIntPipe, Post, Req, UseGuards } from '@nestjs/common';
    import type { Request } from 'express';
    import { JwtAuthGuard } from '../auth/jwt-auth.guard';
    import { RolesGuard } from '../auth/roles.guard';
    import { Roles } from '../auth/roles.decorator';
    import { CurrentUser, type JwtUser } from '../auth/current-user.decorator';
    import { OrgContextGuard } from '../common/org-context/org-context.guard';
    import { OrgContext } from '../common/org-context/org-context.decorator';
    import type { OrgContextDto } from '../common/org-context/org-context.types';
    import { TransfersService } from './transfers.service';
    import { CreateTransferDto } from './dto/create-transfer.dto';
    import { RejectTransferDto } from './dto/reject-transfer.dto';
    import { AdminCreateTransferDto } from './dto/admin-create-transfer.dto';

    @Controller('transfers')
    @UseGuards(JwtAuthGuard)
    export class TransfersController {
      constructor(private readonly service: TransfersService) {}

      @Post()
      create(@Body() dto: CreateTransferDto, @CurrentUser() user: JwtUser, @Req() req: Request) {
        return this.service.create(dto, user, req);
      }

      @Get('incoming')
      @UseGuards(OrgContextGuard)
      listIncoming(@OrgContext() ctx: OrgContextDto | null) {
        if (!ctx) throw new BadRequestException('Header X-Org-Context requerido');
        return this.service.listIncoming(ctx);
      }

      @Post(':id/accept')
      accept(@Param('id', ParseIntPipe) id: number, @CurrentUser() user: JwtUser, @Req() req: Request) {
        return this.service.accept(id, user, req);
      }
