    **3. `activateOrderItems` (línea 149-202) — REFACTOR a `prisma.$transaction`:**

    Convertir el actual `Promise.all(updates)` en un `prisma.$transaction([...operations])`. La razón es **D-08**: el increment de `creditsUsed` debe ser atómico con la activación de ítems.

    **CONSISTENCIA CROSS-MÓDULO:** Para obtener la sub activa usar `this.subscriptions.getActiveForOwner(...)` (mismo helper que OrdersService consume). El update del increment se ejecuta como op dentro del `$transaction` usando el `sub.id` obtenido. Esto unifica el patrón de acceso a Subscription en el codebase v2.

    ```typescript
    private async activateOrderItems(order: {
      id: number;
      userId: number;
      orgId: number | null;
      items: Array<{
        type: string;
        days: number;
        subtotal: number;
        eventId: number | null;
        spotId: number | null;
        heroId: number | null;
        articleId: number | null;
        unitPrice: number;
      }>;
    }) {
      const expirationDate = (days: number) => {
        const d = new Date();
        d.setUTCDate(d.getUTCDate() + days);
        return d;
      };

      const ops: Prisma.PrismaPromise<unknown>[] = [];
      let creditConsumed = 0;

      for (const item of order.items) {
        // Defensive guard: items SUBSCRIPTION no se activan aquí (van por /subscriptions/confirm en Plan 12-04)
        if (item.type === OrderItemType.SUBSCRIPTION) continue;

        if (item.type === OrderItemType.EVENT && item.eventId) {
          ops.push(this.prisma.event.update({
            where: { id: item.eventId },
            data: {
              status: PublicationStatus.PENDING_MODERATION,
              expirationDate: expirationDate(item.days),
            },
          }));
          // D-08: si el EVENT fue cubierto por crédito (unitPrice === 0 && subtotal === 0), incrementar creditsUsed
          if (item.unitPrice === 0 && item.subtotal === 0) {
            creditConsumed += 1;
          }
        } else if (item.type === OrderItemType.SPOT && item.spotId) {
          ops.push(this.prisma.spot.update({
            where: { id: item.spotId },
            data: { status: PublicationStatus.PENDING_MODERATION, days: item.days, amount: item.subtotal, expirationDate: expirationDate(item.days) },
          }));
        } else if (item.type === OrderItemType.HERO && item.heroId) {
          ops.push(this.prisma.hero.update({
            where: { id: item.heroId },
            data: { status: PublicationStatus.PENDING_MODERATION, days: item.days, amount: item.subtotal, expirationDate: expirationDate(item.days) },
          }));
        } else if (item.type === OrderItemType.ARTICLE && item.articleId) {
          ops.push(this.prisma.article.update({
            where: { id: item.articleId },
            data: { status: PublicationStatus.PENDING_MODERATION },
          }));
        }
      }

      // D-08: añadir el incremento al transaction si aplica.
      // Usa SubscriptionsService.getActiveForOwner para coherencia con OrdersService
      // (mismo entry point cross-módulo a la sub viva).
      if (creditConsumed > 0) {
        const sub = await this.subscriptions.getActiveForOwner(
          order.orgId ? null : order.userId,
          order.orgId,
        );
        if (sub) {
          ops.push(this.prisma.subscription.update({
            where: { id: sub.id },
            data: { creditsUsed: { increment: creditConsumed } },
          }));
        }
        // si sub no existe ya (edge case: cycle expired entre add-to-cart y pay), saltamos increment
      }

      await this.prisma.$transaction(ops);
    }
    ```

    **NOTA sobre consistencia:** El advisor (revisión) confirmó que usar `this.subscriptions.getActiveForOwner()` es preferible a la query Prisma directa porque (a) `SubscriptionsModule` ya se importa en `PaymentsModule` para este propósito; (b) encapsula la regla "CANCELLED con cycleEnd futuro sigue válida" en un solo lugar; (c) mantiene paridad con el patrón usado por `OrdersService.getActiveSub`.
