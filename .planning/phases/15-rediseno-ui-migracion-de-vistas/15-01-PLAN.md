    ```tsx
    "use client";
    import { useState } from "react";
    import { useRouter } from "next/navigation";
    import { toast } from "sonner";
    import { Poster } from "./Poster";
    import type { EventItem } from "@/lib/data";

    export function EventCard({ e, today = false }: { e: EventItem; today?: boolean }) {
      const router = useRouter();
      const [saved, setSaved] = useState(false);
      const handleSave = () => {
        setSaved(s => {
          const next = !s;
          if (next) toast.success("Evento guardado", { description: "Lo encuentras en /cuenta/favoritos" });
          else toast.info("Quitado de favoritos");
          return next;
        });
      };
      return (
        <div className="card" onClick={() => router.push(`/evento/${e.slug ?? e.id}`)}>
          <Poster e={e} today={today} saved={saved} onSave={handleSave} />
        </div>
      );
    }
    ```

    "use client";
    import { useEffect, useState } from "react";
    import Link from "next/link";
    import type { HeroSlide } from "@/lib/api";

    export function HeroCarousel({ slides }: { slides: HeroSlide[] }) {
      const [idx, setIdx] = useState(0);
      useEffect(() => {
        if (slides.length <= 1) return;
        const id = setInterval(() => setIdx(i => (i + 1) % slides.length), 7000);
        return () => clearInterval(id);
      }, [slides.length]);
      const prev = () => setIdx(i => (i - 1 + slides.length) % slides.length);
      const next = () => setIdx(i => (i + 1) % slides.length);
      // Estructura: .pcar > .pcar-track[transform: translateX(-idx*100%)] > slides[.pcar-slide]
      // Cada slide: grid 2 cols (texto + arte) con .hc-text y .hc-art
      // Arrows: .pcar-arrows con botones prev/next
      // Dots: .pcar-dots con buttons que setIdx(i)
      return (
        <section className="pcar">
          <div className="pcar-track" style={{ transform: `translateX(-${idx * 100}%)` }}>
            {slides.map((s, i) => (
              <div key={i} className="pcar-slide">
                {/* hc-text con eyebrow, h1, descripción, CTA Link */}
                {/* hc-art con imagen del slide */}
              </div>
            ))}
          </div>
          <div className="pcar-arrows">
            <button onClick={prev}>‹</button>
            <button onClick={next}>›</button>
          </div>
          <div className="pcar-dots">
            {slides.map((_, i) => (
              <button key={i} className={i === idx ? "on" : ""} onClick={() => setIdx(i)} />
            ))}
          </div>
        </section>
      );
