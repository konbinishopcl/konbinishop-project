    Add a new `useEffect` that runs on mount (and when `token` changes) to fetch saved settings and patch the `plan` state. Place it right after the existing `loadSubs` useEffect (after line 99). The settings endpoint returns an array of `{ key, value }` rows. Filter to the 4 plan keys and patch.

    Add:
    ```typescript
    useEffect(() => {
      if (!token) return;
      fetch("/api/settings", { headers: { Authorization: `Bearer ${token}` } })
        .then((r) => (r.ok ? r.json() : null))
        .then((data) => {
          if (!Array.isArray(data)) return;
          const keys: (keyof PlanSettings)[] = [
            "sub_monthly_price",
            "sub_credits_per_month",
            "sub_aviso_discount",
            "sub_portada_discount",
          ];
          const patch: Partial<PlanSettings> = {};
          data.forEach(({ key, value }: { key: string; value: string }) => {
            if (keys.includes(key as keyof PlanSettings)) {
              patch[key as keyof PlanSettings] = value;
            }
          });
          if (Object.keys(patch).length) setPlan((p) => ({ ...p, ...patch }));
        })
        .catch(() => {});
    }, [token]);
    ```
    `useEffect` is already imported (line 2). Do NOT touch the existing `savePlan` function or the hardcoded defaults in the initial `useState` вЂ” they act as fallbacks until settings load.
